// Efficient string matching algorithm
#include <iostream>
#include <string>
#include <vector>

std::vector<int> KMPSearch(const std::string& pattern, const std::string& text) {
    std::vector<int> lps(pattern.length(), 0);
    std::vector<int> matches;
    
    // Compute longest prefix suffix values
    computeLPSArray(pattern, lps);
    
    int i = 0, j = 0;
    while (i < text.length()) {
        if (pattern[j] == text[i]) {
            i++;
            j++;
        }
        if (j == pattern.length()) {
            matches.push_back(i - j);
            j = lps[j - 1];
        } else if (i < text.length() && pattern[j] != text[i]) {
            if (j != 0)
                j = lps[j - 1];
            else
                i++;
        }
    }
    return matches;
}
